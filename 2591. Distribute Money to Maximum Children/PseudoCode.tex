\begin{algorithm}
    \caption{Distribute Money}
    \begin{algorithmic}
        \If {$children > money$}
        \State return $-1$
        \EndIf
        \State $high \leftarrow children \times 8$
        \If {$high > money$}
        \State return $children - 1$
        \EndIf
        \If {$high = money$}
        \State return $children$
        \EndIf
        \If {$high - money = 4$}
        \State return $children - 2$
        \EndIf
        \State $children \leftarrow children - (high - money) / 7$
        \If {$(high - money) \pmod 7 = 0$}
        \State return $children$
        \Else
        \State return $children - 1$
        \EndIf
    \end{algorithmic}
\end{algorithm}
